export default class RequestFeedback extends Component {
 
  constructor(props) {
    super(props);
    this.state = {
 
      searchTerm: '',
      selectedFriendId: [],
 
    }
  }
 
  searchUpdated(term) {
    this.setState({ searchTerm: term })
  }
 
  onCheckBoxPress(id) {
    let tmp = this.state.selectedFriendId;
 
    if (tmp.includes(id)) {
      tmp.splice(tmp.indexOf(id), 1);
    } else {
      tmp.push(id);
    }
 
    this.setState({
      selectedFriendId: tmp,
 
    });
  }
